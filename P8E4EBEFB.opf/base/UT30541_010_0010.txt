འཛུམ་མོ་སྲིན་ལག་རྡོ་རྗེའི་ར་ལྟར་བྱས་པ་སྟེ། ཕྱག་རྒྱ་འདི་ནི་ཕྱག་རྒྱ་བཟང་པོ་ཞེས་བཤད་པས་ལས་ཐམས་ཅད་བྱེད་དོ། །དེ་བཞིན་དུ་ཟད་པར་རྣམས་གྲི་གུག་གི་ཕྱག་རྒྱ་གུག་སྟེ། ཁུ་ཚུར་ལ་མཐེའུ་ཆུང་བསྐྱང་པའོ། །ལྷ་རྣམས་རལ་གྲིའི་ཕྱག་རྒྱས་དགུག་སྟེ། ལག་པས་གཡས་ཀྱི་མཐེའུ་སྲིན་ལག་མཐེ་བོས་མནན་ནས། གུང་མོ་དང་འཛུབ་མོ་གཤིབས་ཏེ་བརྐྱང་པའོ། །གཟའ་རྣམས་ལ་རིན་པོ་ཆེའི་ཕྱག་ནི་གུང་མོ་གཉིས་ནང་དུ་བཀུག་ཅིང་མཐེ་བོང་དང་སྤྲད་ལ་གཞན་རྩེ་སྤྲད་ཅིང་བརྐྱང་པའོ། ཀླུ་རྣམས་ལ་པད་མའི་ཕྱག་རྒྱ་ནི། ལག་པ་གཡོན་པས་མཐེ་ཆུང་མཐེ་བོ་མཉམ་དུ་བྱས་ལ། གཞན་གསུམ་པོ་སོ་སོར་ཕྱེ་བ་ནི་པད་མའོ། །འབྱུང་པོ་ཡི་དྭགས་རྣམས་ནི་འཁོར་ལོའི་ཕྱག་རྒྱ་སྟེ། ལག་པ་གཉིས་ཁ་སྦྱར་ལ་སོར་མོ་རྣམས་བརྐྱང་བའོ། །ཁྱིའི་ཕྱག་རྒྱ་ནི་གྲོང་གི་འབྱུང་པོ་དགུ་སྟེ། དེ་ནས་ཁྲོ་བོ་རྣམས་ལ་སྔགས་འདིས་གཏོར་མ་འབུལ་ལོ། །ཨོཾ་ཨཱཿཧཱུྃ་ཧོ། ཨུཤྣི་ཤ། སུྃ་བྷ་ནི་སུྃ་བྷ། བིགྒ་ནན་ཏཱ་ཀཱ། ནི་ལ་དཱན་དྷ། པྲ་ཛྙ་ཏ་ཀ། ཌ་ཀི་ར་ཙ། པད་མཱན་ཏ་ཀཱ། ཨཱ་ཙ་ལ་ཡ་མན་ཏ་ཀཱ། མ་ཧ་པ་ལཱ་བྷྱ། ས་པ་རི་བྷ་རི་བྷྱཿ ཨི་དྷཾ་བྷ་ལི། གྷན་དྷྃ། པུ་པ། དྷུ་པྃ། དི་པཾ་ཨ་ཀྵ་ཏྃ། ད་དྷ་མ་ཧི། ཏེ་ཙ་ཨ་ཀད་དྷ་ཡཱ། ས་པ་རི་བཱ་ར་ཤྲི་གྷྲྃ། ཨི་དྷྃ་བྷ་ལིང་གྷྲི་ནྣ་ཏུ་ཁ་དན་དུ། པི་པ་དེ་ཛ་ཧཱུྃ་བྃ་ཧོཿ སན་ཧྲི་པ་ཏ། སརྦ་སྭ་ཏ་ནྃ། ཤན་ཏྱྃ། པུས་ཏིཾ། རཀྵ་བྷ་རན། གུར་བྷ་ཏི་ཀུ་བྷན་ཏུ་ཧཱུྃ་ཧཱུྃ་ཕཊ་ཕཊ། བཛྲ་དྷ་ར་ཨཱ་ཛཱ་པ་ཡ་ཏི་སྭ་ཧཱ། ཞེས་མཆོད་པའི་ཁྱད་པར་དང་བདུད་རྩིར་བསམས་ལ་འཕྲོ་བའི་སྦྱོར་བས་ཚིམ་པར་བྱའོ། །ཟད་པར་ལ། ཨོཾ་ཨཱཿཧཱུྃ་ཧོ་ཨ་ཀས་བྷ་ཡུ་ཏེ་ཛ་ཡུ་ཏ་ཀ་པི་ཐིར་བི་པ་ཏི་ཏེ་བྷྱ། ས་པ་རི་བྷ་ར་མན་ཆད་འདྲ་བས་གཏོར་མ་བཏང་བར་བྱའོ། །དེ་ནས་ཕྱོགས་སྐྱོང་ལ། ཨོཾ་ཨཱཿཧཱུྃ་ཧོཿབྲ་ཧྨ་བྷི་ཤཱུ། ཤི་རཱི་ཏེ། བྷ་ཡུ། ཡཱ་མཱ། ཨགྣེ། ས་མུ་ཏྲི། ཨི་ཤཱ་ར། ཨིན་ཏྲ། ཡ་ཀྵེ་བྷྱ།

ས་པ་རི་བྷ་ར་མན་ཆད་འདྲ་བས་གཏོར་མ་སྦྱིན་ནོ། །དེ་ནས་གཟའ་རྣམས་ལ། ཨོཾ་ཨཱཿཧཱུྃ་ཧོ། ར་ཧཱུ་ཀ་ལ་ཨག་ནེ། ཙན་ཏྲ། སུརྻ། བྷུ་དྷྱ། མྃ་གྷ་ལ། ཤུ་ཏྲ། བྷྲི་ཧ་པ་ཏི། ཀེ་ཏུ་ཤ་ནཱ་བྷྱཿ ས་པ་རི་བྷ་ར་མན་ཆད་འཐུན་པས་གཏོར་མ་སྦྱིན་ནོ། དེ་ནས་ཀླུ་རྣམས་ལ། ཨོཾ་ཨཱཿཧཱུྃ་ཧོ་ཛཱ་ཡཱ། བྷི་ཛ་ཡཱ། པད་མ་ཀར་ཀོ་ཏྲ། བྷ་སུ་ཀཱི་ཤི་པ་ལཱ། ཀུ་ལི་ཀཱ། ཨ་ནན་ཏཱ། ཏ་ཀྵ་ཀཱ། མ་ཧ་པད་མེ་བྷྱ། ས་པ་རི་བྷ་ར་གོང་བཞིན་དུ་གཏོར་མ་སྦྱིན་ནོ། །དེ་ནས་འབྱུང་པོ་རྣམས་ལ་ཨོཾ་ཨཱཿཧཱུྃ་ཧོ། བེ་ཏ་ཊ་བཱི་ཀྲི་ཏེ། མུ་ཁཱ་ཤ་ཀ། ཁ་ཤུ་གར། གྷྲིད་ཏ། བྷྱ་གྷྲ། ཨུ་ལུ་ཀ། ཛཱྃ་བུ་ཀ། གྷཱ་རུ་ཊཱ། མུ་ཁྭེ་བྷྱཿ ས་པ་རི་བྷ་ར་མན་ཆད་མཐུན་པས་གཏོར་མ་བྱིན་ལ། བཟོད་པར་གསོལ་ཞིང་གཤེགས་སུ་གསོལ་ཏེ། བསྡུ་བའི་ཚེ། ཁྲོ་བོ་རྣམས་གསང་བར་ཧྃ་ལ་བསྡུ་བའོ། །སྤྱི་གཙུག་ཨྃ་ལས་ཟད་པར་རྣམས་སོ། །དཔྲལ་བར་ཨི་ལ་ཕྱོགས་སྐྱོང་ངོ༌། །མགྲིན་པར་རི་ལ་གཟའ་རྣམས་སོ། །སྙིང་ཁར་ཨུ་ལ་ཀླུ་རྣམས་སོ། །ལྟེ་བར་ལཱི་ལ་འབྱུང་པོ་རྣམས་བསྡུའོ། །བགེགས་ཀྱི་གཏོར་མ་སྦྱང་བའི་སྔགས་དང༌། གཞི་བདག་གི་གཏོར་མ་སྦྱང་བའི་སྔགས་ནི་དབང་ཆོག་གི་དེ་ཉིད་ཡིན་ནོ། །ཡང་བྱེ་བྲག་ཏུ་མཁའ་འགྲོ་མ་རྣམས་དང་དཀྱིལ་འཁོར་གྱི་ལྷ་ལ་ནི། སྔར་ལྟར་གཏོར་མ་བྱིན་གྱིས་བརླབས་ལ་འགྲོན་སྤྱན་དྲང་བའི་ཚེ། སངས་རྒྱས་ཡབ་དང་ཡུམ་དག་བཅས། །སྲས་བཅས་འཁོར་དང་གྲོགས་དང་བཅས། །གཉེན་ནི་མ་ལུས་པས་བསྐོར་བ། །ལྷ་དང་ལྷ་མོ་དག་གི་ཚོགས། །རྣམས་ཀྱི་དགེ་བར་མཛད་བཞིན་པ། །རྡོ་རྗེ་སེམས་དཔའ་འདིར་སྤྱོན་ཅིག ཅེས་བརྗོད་ནས་རྡོ་རྗེ་མཁའ་འགྲོ་མ་རྣམས་ནི། ཨོཾ་ཧ་ཧཱ་ཧི་ཧཱི་བཛྲ་བྷ་རེ་བཱ། ཨ་ཀར་ཤ་ཡཱ། པྲ་བེ་ཤ་ཡཱ་པྲ་བེ་ཤ་ཡཱ། བྷན་དྷ་ཡཱ་བྷན་དྷ་ཡཱ། ཏོ་ཤ་ཡཱ་ཏོ་ཤ་ཡཱ། ཛ་ཧཱུྃ་བཾ་ཧོ། བཛྲ་དྷ་ཀི་ནི་ཧྲི་དྷ་ཡྃ་ཧཱུྃ་ཧཱུྃ་ཕྲེཾ་ཕྲེཾ་ཕཊ་ཞེས་བཀུག་ནས་མཆོད་པའི་ལྷ་མོ་རྣམས་ཀྱིས་མཆོད་ཅིང༌། ཨོཾ་ཨཱཿཧཱུྃ་ཧོ།

བཛྲ་དྷ་ཀི་ནི་སརྦ་བྷྱ་ཡ་ས་པ་རི་ཝ་རི་བྷྱ་མན་ཆད་ཐམས་ཅད་འདྲ། དེ་ནས་བཟོད་པར་གསོལ་ཞིང་གཤེགས་སུ་གསོལ་ཏེ། །སྨོན་ལམ་གདབ་བོ། །གཏོར་མའི་ཆོ་ག་རྫོགས་སོ།། །།
